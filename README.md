    import math
    def dbetn(a,b):
        x1,y1=a
        x2,y2=b
        distance=math.sqrt((x2-x1)**2+(y2-y1)**2)
        return round(distance,2)
    a=[10,6]
    b=[4,2]
    c=dbetn(a,b)
    print(c)
